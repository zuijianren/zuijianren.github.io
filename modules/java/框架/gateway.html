ModifyRequestBody ​

仅允许通过 Java DSL(Domain-specific language 领域特定语言) 进行配置

java
@Bean
public RouteLocator routes(RouteLocatorBuilder builder) {
    return builder.routes()
        .route("rewrite_request_obj", r -> r.host("*.rewriterequestobj.org")
            .filters(f -> f.prefixPath("/httpbin")
                .modifyRequestBody(String.class, Hello.class, MediaType.APPLICATION_JSON_VALUE,
                    (exchange, s) -> return Mono.just(new Hello(s.toUpperCase())))).uri(uri))
        .build();
}

static class Hello {
    String message;

    public Hello() { }

    public Hello(String message) {
        this.message = message;
    }

    public String getMessage() {
        return message;
    }

    public void setMessage(String message) {
        this.message = message;
    }
}

自定义 Route Predicate Factories ​

创建 CheckAuthRoutePredicateFactory 类 ​

java
@Slf4j
@Component
public class CheckAuthRoutePredicateFactory extends AbstractRoutePredicateFactory<CheckAuthRoutePredicateFactory.Config> {


    public CheckAuthRoutePredicateFactory() {
        super(Config.class);
    }

    /**
     * 返回当前请求是否匹配这个请求头
     *
     * @param config 自定义 Predicate 配置
     * @return Predicate 结果
     */
    @Override
    public Predicate<ServerWebExchange> apply(CheckAuthRoutePredicateFactory.Config config) {
        // 从 config 对象中获取配置
        return exchange -> {
            // 获取 request
            ServerHttpRequest request = exchange.getRequest();
            // 从请求中获取信息 判断是否与配置匹配
            String auth = request.getHeaders().getFirst("auth");
            // 校验 auth 是否匹配
            if (config.getName().equals(auth)) {
                return true;
            }
            return false;
        };
    }


    // 如果需要 还可以添加参数校验
    public static class Config {

        // 配置项...

        private String name;

        public String getName() {
            return name;
        }

        public void setName(String name) {
            this.name = name;
        }
    }
}

自定义 Route GatewayFilter Factory ​

创建 PreGatewayFilterFactory 类 ​

java
@Component
public class PreGatewayFilterFactory extends AbstractGatewayFilterFactory<PreGatewayFilterFactory.Config> {


    private static final String HEADER_NAME = "auth";

    public PreGatewayFilterFactory() {
        super(Config.class);
    }

    @Override
    public GatewayFilter apply(Config config) {
        // 从 config 类中获取配置
        return (exchange, chain) -> {
            String auth = exchange.getRequest().getHeaders().getFirst(HEADER_NAME);
            // 如果想要对请求进行 预处理  则需要在 chain.filter 前执行
            ServerHttpRequest.Builder builder = exchange.getRequest().mutate();
            // 修改 auth 请求头 例如: 'lisi' => 'super lisi'
            builder.header(HEADER_NAME, config.getPrefix() + " " + auth);
            // 使用 builder 来修改请求
            return chain.filter(exchange.mutate().request(builder.build()).build());
        };
    }

    public static class Config {

        private String authPrefix;

        public String getAuthPrefix() {
            return authPrefix;
        }

        public void setAuthPrefix(String authPrefix) {
            this.authPrefix = authPrefix;
        }
    }
}

创建 PostGatewayFilterFactory 类 ​

java
@Component
public class PostGatewayFilterFactory extends AbstractGatewayFilterFactory<PostGatewayFilterFactory.Config> {


    public PostGatewayFilterFactory() {
        super(Config.class);
    }

    @Override
    public GatewayFilter apply(Config config) {
        // 从 config 类中获取配置
        return (exchange, chain) -> {
            return chain.filter(exchange).then(Mono.fromRunnable(() -> {
                ServerHttpResponse response = exchange.getResponse();
                // 处理相应结果
                response.setRawStatusCode(config.getResponseStatus());
            }));
        };
    }

    @Valid
    public static class Config {

        @NotNull
        private Integer responseStatus;

        public Integer getResponseStatus() {
            return responseStatus;
        }

        public void setResponseStatus(Integer responseStatus) {
            this.responseStatus = responseStatus;
        }
    }
}

自定义全局过滤器 ​

创建 CustomGlobalFilter 类 ​

java
@Slf4j
@Component
public class CustomGlobalFilter implements GlobalFilter {

    @Override
    public Mono<Void> filter(ServerWebExchange exchange, GatewayFilterChain chain) {

        log.info("custom global filter");
        // 通过 exchange 对象 获取 route 配置的 metadata
        // 尝试访问: http://localhost:8080/system/hello
        Route routeWithMetadata = exchange.getAttribute(GATEWAY_ROUTE_ATTR);
        assert routeWithMetadata != null;
        Map<String, Object> metadata = routeWithMetadata.getMetadata();
        log.info("route: metadata: ");
        metadata.forEach((key, value) -> log.info("{} : {}", key, value));
        return chain.filter(exchange);
    }
}

配置类中创建匿名全局过滤器 ​

java
@Configuration
public class GatewayConfig {

    @Bean
    public GlobalFilter customGlobalBeforeFilter() {
        return (exchange, chain) -> exchange.getPrincipal()
                .map(Principal::getName)
                .defaultIfEmpty("Default User")
                .map(userName -> {
                    //adds header to proxied request
                    exchange.getRequest().mutate().header("CUSTOM-REQUEST-HEADER", userName).build();
                    return exchange;
                })
                .flatMap(chain::filter);
    }

    @Bean
    public GlobalFilter customGlobalPostFilter() {
        return (exchange, chain) -> chain.filter(exchange)
                .then(Mono.just(exchange))
                .map(serverWebExchange -> {
                    //adds header to response
                    serverWebExchange.getResponse().getHeaders().set("CUSTOM-RESPONSE-HEADER",
                            HttpStatus.OK.equals(serverWebExchange.getResponse().getStatusCode()) ? "It worked": "It did not work");
                    return serverWebExchange;
                })
                .then();
    }
}
